# phase-0-gps-1
git clone - clones a repo from a remote to a local source
git add - adds file to the stage
git commit - commits the changes
git push - pushes to github
git remote - manages functions of remotes
rm - removes a file
touch (New-Item filename -type file) - creates a new file
git checkout - creates/manages branches
cd - changes directory
ls - lists all files in a directory
